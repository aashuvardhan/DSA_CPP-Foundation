#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	vector<int> v={34, 8, 10, 3, 2, 80, 30, 33, 1};
	
	vector<int> left(v.size());
	vector<int> right(v.size());
	
	left[0]=v[0];
	
	for(int i=1;i<v.size();i++){
	    left[i]=min(left[i-1],v[i]);
	}
	
	
	for(auto ele: left){
	    cout<<ele<<"  ";
	}
	cout<<endl;
	int n=v.size();
	right[n-1]=v[n-1];
	for(int i=n-2;i>=0;i--){
	    right[i]=max(right[i+1],v[i]);
	}
	
	for(auto ele: right){
	    cout<<ele<<"  ";
	}
    cout<<endl;
    
    int i=0;
    int j=0;
    
    int res=0;
    while(i<n and j<n){
        if(left[i]<=right[j]){
            res=max(res,j-i);
            j++;
        }else{
            i++;
        }
    }
    cout<<res;
    
}
