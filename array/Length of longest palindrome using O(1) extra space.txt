/*


'''
Length of longest palindrome using O(1) extra space

Input  : List = 2,3,7,3,2,12,24
Output : 5
The longest palindrome list is 2,3,7,3,2


'''


*/



#include <bits/stdc++.h>
using namespace std;

int counter(vector<int>& v,int start,int end,int n){
    
    int count=0;
    while(start>=0 and end<n){
        if(v[start]==v[end]){
            count++;
        }else{
            break;
        }
        start--;
        end++;
    }
    return count;
    
}



int main() {
	// your code goes here
	
	vector<int> v={2,3,7,3,2,12,24};
	
	int n=v.size();
	int res=0;
	for(int i=0;i<n;i++){
	    
	    //odd
	    res=max(res,2*counter(v,i-1,i+1,n)+1);
	    //even
	    res=max(res,2*counter(v,i,i+1,n));
	    
	    
	}
	cout<<res;
	

}
