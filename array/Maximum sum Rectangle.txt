#include <bits/stdc++.h>
using namespace std;


int kadane(vector<int>& v){
    int n=v.size();
    
    int maxtill=v[0];
    int maxhere=v[0];
    
    for(int i=1;i<n;i++){
        maxhere=max(maxhere+v[i],v[i]);
        maxtill=max(maxtill,maxhere);
    }
    
    return maxtill;
}


int main() {
    vector<vector<int>> v={
    { 5,  5},
    {-10, -10}
};
    
    int row=v.size();
    int col=v[0].size();
    
    // FIX 1: Initialize to smallest possible integer to handle all-negative matrices
    int maxi=INT_MIN; 
    
    // approach: fix the top row and move the bottom row
    for(int i=0;i<row;i++){
        vector<int> temp(col,0);
        
        for(int j=i;j<row;j++){
            
            // Update the 1D array with the current row's column values
            for(int k=0;k<col;k++){
                temp[k]+=v[j][k];
            }
            
            // FIX 2: Evaluate Kadane's INSIDE the j-loop to check every top/bottom pair
            maxi=max(maxi, kadane(temp)); 
        }
    }
    
    cout<<maxi;
    return 0;
}
