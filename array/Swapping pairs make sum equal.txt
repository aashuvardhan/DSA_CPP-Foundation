/*
'''
Given two arrays of integers A[] and B[] of size N and M, the 
task is to check if a pair of values (one value from each array) 
exists such that swapping the elements of the pair will make the 
sum of two arrays equal.

 

Example 1:

Input: N = 6, M = 4
A[] = {4, 1, 2, 1, 1, 2}
B[] = (3, 6, 3, 3)
Output: 1
Explanation: Sum of elements in A[] = 11
Sum of elements in B[] = 15, To get same 
sum from both arrays, we can swap following 
values: 1 from A[] and 3 from B[]
Example 2:

Input: N = 4, M = 4
A[] = {5, 7, 4, 6}
B[] = {1, 2, 3, 8}
Output: 1
Explanation: We can swap 6 from array 
A[] and 2 from array B[]

'''



*/



#include <bits/stdc++.h>
using namespace std;



int main(){
    
    
    vector<int> a={4, 1, 2, 1, 1, 2};
    vector<int> b={3, 6, 3, 3};
    
    int suma=0;
    int sumb=0;
    
    for(auto ele: a){
        suma+=ele;
    }
    for(auto ele: b){
        sumb+=ele;
    }
    
    if(suma==sumb){
        cout<<1;
    }else{
        if((suma+sumb)%2!=0){
            cout<<0;
        }else{
            int target=(suma+sumb)/2;
            int extraA=suma-target;
            int extraB=sumb-target;
            
            if(sumb>suma){
                unordered_set<int> sta;
                for(auto ele: a){
                    sta.insert(ele);
                }
                int flag=0;
                for(auto ele: b){
                    if(sta.find(ele-extraB)!=sta.end()){
                        cout<<1;
                        flag=1;
                        break;
                    }
                }
                if(flag==0){
                    cout<<0;
                }
            }else{
                unordered_set<int> stb;
                for(auto ele: b){
                    stb.insert(ele);
                }
                
                int flag=0;
                for(auto ele: a){
                    if(stb.find(ele-extraA)!=stb.end()){
                        cout<<1;
                        flag=1;
                        break;
                    }
                }
                if(flag==0){
                    cout<<0;
                }
            }
            
            
        }
    }
    
    
}
