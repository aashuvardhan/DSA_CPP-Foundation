/*

'''
Given a string s, find the length of the longest substring without repeating characters.

Input: s = "abcabcbb"
Output: 3
Explanation: The answer is "abc", with the length of 3.

'''


*/



#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	string str="abcabcbb";
	int n=str.size();
	vector<int> visited(256,-1);
	
	int res=0;
	int j=0;
	
	for(int i=0;i<n;i++){
	    int val=(int)str[i];
	    
	    j=max(j,visited[val]+1);
	    
	    res=max(res,i-j+1);
	    
	    visited[val]=i;
	    
	    
	}
	cout<<res;
	

}
