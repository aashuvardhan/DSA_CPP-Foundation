#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	
	vector<int> v={2,3,2,4,4,6,3,2};
	vector<int> left(v.size());
	
	left[0]=0;
	int left_INCR=0;
	for(int i=1;i<v.size();i++){
	    if(v[i]>v[i-1]){
	        left_INCR=i;
	    }
	    left[i]=left_INCR;
	}
	
	
	
	vector<int> right(v.size());
	int n=v.size();
	right[n-1]=n-1;
	int right_INCR=n-1;
	
	for(int i=n-2;i>=0;i--){
	    if(v[i]>v[i+1]){
	        right_INCR=i;
	    }
	    right[i]=right_INCR;
	}
	
	
	//single query
	int start=3;
	int end=5;
	
	if(right[start]>=left[end]){
	    cout<<"Yes";
	}
	

}
