#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	
	vector<int> v={2,3,2,4,4,6,3,2};
	int n=v.size();
	vector<int> left(n);
	left[0]=0;
	int ptr_left=0;
	for(int i=1;i<n;i++){
	    //this means that the peak is reached after increasing at i if we start from left side compared to the previous element.
	    if(v[i]>v[i-1]){
	        ptr_left=i;
	    }
	    left[i]=ptr_left;
	}
	
	vector<int> right(n);
	right[n-1]=n-1;
	int ptr_right=n-1;
	for(int i=n-2;i>=0;i--){
	    //this means that this i was the peak and now decresing will start towards right compared to the previous element.
	    if(v[i]>v[i+1]){
	        ptr_right=i;
	    }
	    right[i]=ptr_right;
	}
	
	
	vector<vector<int>> query={{3,5},{1,3},{0,2}};
	
	for(auto ele: query){
	    int start=ele[0];
	    int end=ele[1];
	    
	    if(right[start]>=left[end]){
	        //here we are checking that ab jo decreasing start hoga wo incresing ke baad ho.
	        cout<<"yes"<<endl;
	    }else{
	        cout<<"no"<<endl;
	    }
	}
	

}
