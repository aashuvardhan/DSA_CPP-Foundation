#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	string s="0000001";
	int n=s.size();
	vector<int> v;
	
	
	int count=0;
	for(int i=0;i<n;i++){
	    if(s[i]=='0'){
	        count++;
	        v.push_back(1);
	    }else{
	        v.push_back(-1);
	    }
	}
	
	if(count==n){
	    cout<<-1;
	}else{
	    //modified kadane
	    int maxtill=0;
	    int maxhere=0;
	    int initiator=0;
	    int start=0;
	    int end=0;
	    
	    for(int i=0;i<n;i++){
	        maxhere+=v[i];
	        if(maxhere<0){
	            maxhere=0;
	            initiator=i+1;
	        }else if(maxtill<maxhere){
	            maxtill=maxhere;
	            start=initiator;
	            end=i;
	        }
	    }
	    cout<<start<<"  "<<end;
	    
	    
	}
	
	
	

}
