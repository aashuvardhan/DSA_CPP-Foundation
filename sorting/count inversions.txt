#include <bits/stdc++.h>
using namespace std;



int merge(vector<int>& v,int low,int mid,int high){
    
    vector<int> left;
    vector<int> right;
    
    for(int i=low;i<=mid;i++){
        left.push_back(v[i]);
    }
    
    for(int i=mid+1;i<=high;i++){
        right.push_back(v[i]);
    }
    
    
    int i=0;
    int m=left.size();
    int j=0;
    int n=right.size();
    
    int k=low;
    
    int res=0;
    
    while(i<m and j<n){
        if(left[i]<=right[j]){
            v[k]=left[i];
            i++;
        }else{
            res+=(m-i);
            v[k]=right[j];
            j++;
        }
        k++;
    }
    
    while(i<m){
        v[k]=left[i];
        i++;
        k++;
    }
    
    while(j<n){
        v[k]=right[j];
        k++;
        j++;
    }
    
    
    return res;
    
    
}



int mergesort(vector<int>& v,int low,int high){
    
    int res=0;
    if(low<high){
        
        int mid=(low+high)/2;
        res+=mergesort(v,low,mid);
        res+=mergesort(v,mid+1,high);
        res+=merge(v,low,mid,high);
    }
    return res;
}

	int low=0;
	int high=v.size()-1;
	
	int res=mergesort(v,low,high);

}
