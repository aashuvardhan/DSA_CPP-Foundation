#include <bits/stdc++.h>
using namespace std;


int lomuto_partition(vector<int>& v,int low,int high){
    int i=low-1;
    int pivot=v[high];
    for(int j=low;j<high;j++){
        if(v[j]<pivot){
            i++;
            int temp=v[i];
            v[i]=v[j];
            v[j]=temp;
        }
    }
    
    int temp=v[i+1];
    v[i+1]=v[high];
    v[high]=temp;
    return i+1;
}



int main() {
	// your code goes here
	vector<int> v={30,20,5,10,8};
    int n=v.size();
    int k=4;
    
    int low=0;
    int high=n-1;
    int res=-1;
    
    while(low<=high){
        int p=lomuto_partition(v,low,high);
        
        if(p==k-1){
            res=v[p];
            break;
        }else if(p<k-1){
            low=p+1;
        }else{
            high=p-1;
        }
    }
    cout<<res;

}
