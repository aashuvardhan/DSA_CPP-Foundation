/*
https://www.geeksforgeeks.org/dsa/queries-to-find-maximum-product-pair-in-range-with-updates/


You are given an array arr[] of positive integers. Your task is to process q queries, each consisting of three integers, of the following two types:

[1, L, R]: Find the maximum product of any two distinct elements in the subarray arr[L…R] (inclusive).
[2, i, val]: Update the element at index i to val (i.e., set arr[i] = val) where val > 0
Examples:

Input: n = 5, q = 3
arr[] = [1, 3, 4, 2, 5]
queries[][] = [ [1, 0, 2], [2, 1, 6], [1, 0, 2] ]
Output: 12 24
Explanation: For the query 1, the maximum product in a range [0, 2] is 3 * 4 = 12. 
For the query 2, after an update, the array becomes [1, 6, 4, 2, 5] 
For the query 3, the maximum product in a range [0, 2] is 6 * 4 = 24. 



*/


#include <bits/stdc++.h>
using namespace std;


void update(vector<pair<int,int>>& tree,int low,int high,int tree_ind,int pos,int val){
    
    if(low>high){
        return;
    }
    
    if(low==high){
        tree[tree_ind].first=val;
        tree[tree_ind].second=INT_MIN;
        return;
    }
    
    int mid=(low+high)/2;
    if(pos<=mid){
        update(tree,low,mid,2*tree_ind+1,pos,val);
    }else{
        update(tree,mid+1,high,2*tree_ind+2,pos,val);
    }
    pair<int,int> left=tree[2*tree_ind+1];
    pair<int,int> right=tree[2*tree_ind+2];
    
    tree[tree_ind].first=max(left.first,right.first);
    tree[tree_ind].second=min(max(left.first,right.second),max(left.second,right.first));
    
    
}



pair<int,int> range_query(vector<pair<int,int>>& tree,int low,int high,int qstart,int qend,int tree_ind){
    
    if(low>high or qstart>high or qend<low){
        return {-1,-1};
    }
    
    if(qstart<=low and qend>=high){
        return tree[tree_ind];
    }
    
    int mid=(low+high)/2;
    pair<int,int> left=range_query(tree,low,mid,qstart,qend,2*tree_ind+1);
    pair<int,int> right=range_query(tree,mid+1,high,qstart,qend,2*tree_ind+2);
    
    pair<int,int> temp;
    temp.first=max(left.first,right.first);
    temp.second=min(max(left.first,right.second),max(left.second,right.first));
    return temp;
    
    
}



pair<int,int> construct_tree(vector<int>& arr,vector<pair<int,int>>& tree,int low,int high,int tree_ind){
    
    if(low==high){
        tree[tree_ind].first=arr[low];
        tree[tree_ind].second=INT_MIN;
        return tree[tree_ind];
    }
    
    int mid=(low+high)/2;
    pair<int,int> left=construct_tree(arr,tree,low,mid,2*tree_ind+1);
    pair<int,int> right=construct_tree(arr,tree,mid+1,high,2*tree_ind+2);
    
    int largest=max(left.first,right.first);
    int second_largest=min(max(left.first,right.second),max(left.second,right.first));
    
    tree[tree_ind].first=largest;
    tree[tree_ind].second=second_largest;
    return tree[tree_ind];
    
    
    
}


int main() {
	// your code goes here
	vector<int> v={1, 3, 4, 2, 5};
	int n=v.size();
	int low=0;
	int high=n-1;
	
	//the pair store the largest and second largest.
	vector<pair<int,int>> tree(4*n);
	
	construct_tree(v,tree,low,high,0);
	
	vector<vector<int>> queries =  { {1, 0, 2}, {2, 1, 6}, {1, 0, 2}};
	
	for(auto ele: queries){
	    if(ele[0]==1){
	        pair<int,int> res=range_query(tree,low,high,ele[1],ele[2],0);
	        cout<<res.first*res.second<<endl;
	    }else{
	        update(tree,low,high,0,ele[1],ele[2]);
	        v[ele[1]]=ele[2];
	    }
	}
	

}
