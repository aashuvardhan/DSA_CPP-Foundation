using namespace std;

struct Node{
	    int data;
	    Node *next;
	    Node(int x){
	        data=x;
	        next=NULL;
	    }
	};
	
	
Node* deepCopy(Node* head){
    Node* newhead=NULL;
    
    if(head==NULL){
        return newhead;
    }else{
        newhead= new Node(head->data);
    }
    Node* curr=newhead;
    head=head->next;
    
    while(head!=NULL){
        Node* temp=new Node(head->data);
        curr->next=temp;
        curr=curr->next;
        head=head->next;
    }
    return newhead;
    
}
	




void traversal(Node* head){

}

Node* rotate_from_right(Node* head,int k){
    Node* newhead=deepCopy(head);
    
    if(newhead==NULL or k==0){
        return newhead;
    }
    
    
    int length=0;
    Node* curr=newhead;
    
    while(curr!=NULL){
        length++;
        curr=curr->next;
    }
    
    if(k>length){
        k=k%length;
    }
    
    Node* lastptr=NULL;
    int count=1;
    
    Node* curry=newhead;
    Node* prevptr=NULL;
    while(curry!=NULL){
        lastptr=curry;
        if(count==(length-k)){
            prevptr=curry;
        }
        
        curry=curry->next;
        count++;
        
    }
    
    lastptr->next=newhead;
    Node* nextt=prevptr->next;
    prevptr->next=NULL;
    newhead=nextt;
    return newhead;
    
    
}
