/* the problem in cpp is that whatever method you try u cannt create a copy of the 
nodes or Linked List. so we need to manually create a deep copy or parallel copy of the 
nodes in order to preserve the original linked list. any other method tampers the original list.


*/


#include <bits/stdc++.h>
using namespace std;

struct Node{
	    int data;
	    Node *next;
	    Node(int x){
	        data=x;
	        next=NULL;
	    }
	};
	
	
Node* deepCopy(Node* head){
    Node* newhead=NULL;
    
    if(head==NULL){
        return newhead;
    }else{
        newhead= new Node(head->data);
    }
    Node* curr=newhead;
    head=head->next;
    
    while(head!=NULL){
        Node* temp=new Node(head->data);
        curr->next=temp;
        curr=curr->next;
        head=head->next;
    }
    return newhead;
    
}
	
	


Node* removeDuplicate(Node* head){
    Node* newhead=deepCopy(head);
    Node* curr=newhead;
    while(curr!=NULL and curr->next!=NULL){
        if(curr->data==curr->next->data){
            Node* nextt=curr->next;
            curr->next=nextt->next;
            delete(nextt);
        }else{
            curr=curr->next;
        }
    }
    return newhead;
    
}

void traversal(Node* head){
    
    Node* curr=head;
	while(curr!=NULL){
	    cout<<curr->data<<"  ";
	    curr=curr->next;
	}
	cout<<endl;
}
