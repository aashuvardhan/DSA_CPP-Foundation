/* the problem in cpp is that whatever method you try u cannt create a copy of the 
nodes or Linked List. so we need to manually create a deep copy or parallel copy of the 
nodes in order to preserve the original linked list. any other method tampers the original list.


*/

#include <bits/stdc++.h>
using namespace std;

struct Node{
	    int data;
	    Node *next;
	    Node(int x){
	        data=x;
	        next=NULL;
	    }
	};



void traversal(Node* head){
    
    Node* curr=head;
	while(curr!=NULL){
	    cout<<curr->data<<"  ";
	    curr=curr->next;
	}
	cout<<endl;
}



Node* removeDuplicate(Node* head){
    Node* curr=head;
    
    while(curr!=NULL and curr->next!=NULL){
        Node* nextt=curr->next;
        if(curr->data==nextt->data){
            curr->next=nextt->next;
            delete(nextt);
        }else{
            curr=curr->next;
        }
    }
    
    return head;
    
    
}



int main() {
	// your code goes here
	Node* head=new Node(10);
	Node* temp=new Node(20);
	Node* temp1=new Node(30);
	Node* dup=new Node(30);
	Node* temp2=new Node(40);
	Node* temp3=new Node(50);
	
	head->next=temp;
	temp->next=temp1;
	temp1->next=dup;
	dup->next=temp2;
	temp2->next=temp3;
	
	traversal(head);
	
	Node* rmdup=removeDuplicate(head);
	

	traversal(rmdup);
	
	

}
