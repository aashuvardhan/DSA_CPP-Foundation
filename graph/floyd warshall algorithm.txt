#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	vector<vector<int>> matrix= {{0,1,43},{1,0,6},{-1,-1,0}};
	int n=matrix.size();
	for(int i=0;i<n;i++){
	    for(int j=0;j<n;j++){
	        if(matrix[i][j]==-1){
	            matrix[i][j]=INT_MAX;
	        }
	    }
	}
	
	
	for(int k=0;k<n;k++){
	    for(int i=0;i<n;i++){
	        for(int j=0;j<n;j++){
	            matrix[i][j]=min(matrix[i][j],matrix[i][k]+matrix[k][j]);
	        }
	    }
	}
	
	for(int i=0;i<n;i++){
	    for(int j=0;j<n;j++){
	        if(matrix[i][j]==INT_MAX){
	            matrix[i][j]=-1;
	        }
	    }
	}
	

}
