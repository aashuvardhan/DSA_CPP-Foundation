/*

'''
Given a sorted dictionary of an alien language having N words
and k starting alphabets of standard dictionary. Find the order
of characters in the alien language.
Note: Many orders may be possible for a particular test case, 
thus you may return any valid order and output will be 1 if the
order of string returned by the function is correct else 0 denoting 
incorrect string returned.
 

Example 1:

Input: 
N = 5, K = 4
dict = {"baa","abcd","abca","cab","cad"}
Output:
1
Explanation:
Here order of characters is 
'b', 'd', 'a', 'c' Note that words are sorted 
and in the given language "baa" comes before 
"abcd", therefore 'b' is before 'a' in output.
Similarly we can find other orders.
Example 2:

Input: 
N = 3, K = 3
dict = {"caa","aaa","aab"}
Output:
1
Explanation:
Here order of characters is
'c', 'a', 'b' Note that words are sorted
and in the given language "caa" comes before
"aaa", therefore 'c' is before 'a' in output.
Similarly we can find other orders.
 

Your Task:
You don't need to read or print anything. Your task is to complete 
the function findOrder() which takes  the string array dict[], its 
size N and the integer K as input parameter and returns a string 
denoting the order of characters in the alien language.


Expected Time Complexity: O(N * |S| + K) , where |S| denotes maximum length.
Expected Space Compelxity: O(K)


Constraints:
1 ≤ N, M ≤ 300
1 ≤ K ≤ 26
1 ≤ Length of words ≤ 50


'''

#include <bits/stdc++.h>
using namespace std;


void dfs(unordered_map<char,vector<char>>& graph,unordered_set<char>& visited,
	vector<char>& stk,char source){
	    
	    visited.insert(source);
	    for(auto adj: graph[source]){
	        if(visited.find(adj)==visited.end()){
	            dfs(graph,visited,stk,adj);
	        }
	    }
	    
	    stk.push_back(source);
	    
	}


int main() {
	// your code goes here
	vector<string> dict={"baa","abcd","abca","cab","cad"};
	int n=5;
	int k=4;
	
	unordered_set<int> st;
	unordered_map<char,vector<char>> graph;
	
	for(int i=0;i<n-1;i++){
	    
	    int j=0;
	    int k=0;
	    string wrd1=dict[i];
	    string wrd2=dict[i+1];
	    while(j<wrd1.size() and k<wrd2.size()){
	        if(wrd1[j]!=wrd2[k]){
	            graph[wrd1[j]].push_back(wrd2[k]);
	            st.insert(wrd1[j]);
	            st.insert(wrd2[k]);
	            break;
	        }
	        j++;
	        k++;
	    }
	}
	
	for(auto wrd: dict){
	    for(char ch: wrd){
	        if(st.find(ch)==st.end()){
	            st.insert(ch);
	        }
	    }
	    if(st.size()==k){
	        break;
	    }
	}
	
	
	unordered_set<char> visited;
	vector<char> stk;
	for(auto ele: st){
	    if(visited.find(ele)==visited.end()){
	        dfs(graph,visited,stk,ele);
	    }
	}
	
	reverse(stk.begin(),stk.end());
	for(auto ele: stk){
	    cout<<ele<<"   "; 
	}

}
