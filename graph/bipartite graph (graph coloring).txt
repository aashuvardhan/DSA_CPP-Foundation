#include <bits/stdc++.h>
using namespace std;



bool bfs(vector<vector<int>>& graph,vector<int>& color,int source){
    color[source]=1;
    vector<int> queue;
    queue.push_back(source);
    
    while(queue.size()!=0){
        int curr=queue[0];
        queue.erase(queue.begin());
        for(int adj: graph[curr]){
            if(color[adj]==-1){
                color[adj]=1-color[curr];
                queue.push_back(adj);
            }else if(color[adj]==color[curr]){
                
                return false;
            }
        }
    }
    return true;
    
}


bool bipartite(vector<vector<int>>& graph,int n){
    
    vector<int> color(n,-1);
    
    for(int i=0;i<n;i++){
        if(color[i]==-1 and bfs(graph,color,i)==false){
            return false;
        }
    }
    return true;
    
}

int main() {
	// your code goes here
	int n,m;
	cin>>n>>m;
	
	vector<vector<int>> graph(n);
	
	for(int i=0;i<m;i++){
	    int start,end;
	    cin>>start>>end;
	    graph[start].push_back(end);
	    graph[end].push_back(start);
	    
	}
	
	cout<<bipartite(graph,n);
	
	
	
}
