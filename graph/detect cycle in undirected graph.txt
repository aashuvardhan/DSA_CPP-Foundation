#include <bits/stdc++.h>
using namespace std;


bool detect_cyc_und_util(vector<vector<int>>& graph,vector<bool>& visited,int source,int parent){
    
    
    visited[source]=true;
    
    for(auto adj: graph[source]){
        if(visited[adj]==false){
            if(detect_cyc_und_util(graph,visited,adj,source)==true){
                return true;
            }
        }else if(adj!=parent){
            return true;            
        }
    }
    return false;
    
}


bool detect_cyc_und(vector<vector<int>>& graph,int n){
    
    vector<bool> visited(n);
    
    for(int i=0;i<n;i++){
        if(visited[i]==false){
            if(detect_cyc_und_util(graph,visited,i,-1)==true){
                return true;
            }
        }
    }
    return false;
    
    
}


int main() {
	// your code goes here
	int n,m;
	
	cin>>n>>m;
	
	vector<vector<int>> graph(n);
	
	for(int i=0;i<m;i++){
	    int u,v;
	    cin>>u>>v;
	    
	    graph[u].push_back(v);
	    graph[v].push_back(u);
	    
	}
	
	cout<<detect_cyc_und(graph,n)<<endl;

}
