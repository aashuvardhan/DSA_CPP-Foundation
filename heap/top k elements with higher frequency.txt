#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	
	vector<int> v={1,1,1,2,2,3};
	int k=2;
	
	unordered_map<int,int> umap;
	for(int i=0;i<v.size();i++){
	    umap[v[i]]++;
	}
	
	priority_queue<pair<int,int>> pq;
	//max-heap
	
	for(auto ele: umap){
	    pq.push({ele.second,ele.first});
	}
	
	for(int i=0;i<k;i++){
	    cout<<pq.top().second<<endl;
	    pq.pop();
	}

}
