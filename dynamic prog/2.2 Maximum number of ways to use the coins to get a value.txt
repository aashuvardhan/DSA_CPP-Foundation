	int value=5;
	int n=coins.size();
	
	vector<vector<int>> tab(n+1,vector<int>(value+1));
	
	for(int j=0;j<=value;j++){
	    tab[0][j]=0;
	}
	for(int i=0;i<=n;i++){
	    tab[i][0]=1;
	}
	
	for(int i=1;i<=n;i++){
	    for(int j=1;j<=value;j++){
	        
	        if(coins[i-1]<=j){
	            tab[i][j]=tab[i-1][j]+tab[i][j-coins[i-1]];
	        }
	        else{
	            tab[i][j]=tab[i-1][j];
	        }
	    }
	}
	
	cout<<tab[n][value];
	
	

}
