#include <bits/stdc++.h>
using namespace std;
//minimum number of partitions that we can make in the given string so that every substring is a palindrome.
bool palindrome(string str,int i,int j){
    
    while(i<=j){
        if(str[i]!=str[j]){
            return false;
        }else{
            i++;
            j--;
        }
        
    }
    return true;
    
}

int solve(string str,int i,int j){
    
    if(i>=j){
        return 0;
    }
    if(palindrome(str,i,j)){
        return 0;
    }
    
    int mini=INT_MAX;
    for(int k=i;k<=j-1;k++){
        if(palindrome(str,i,k)){
            int tempans=1+solve(str,k+1,j);
            mini=min(mini,tempans);
        }
    }
    return mini;
    
    
}

	cout<<solve(str,0,n-1);
