	int capacity=10;
	int n=weights.size();
	vector<vector<int>> tab(n+1,vector<int>(capacity+1));
	
	for(int j=0;j<=capacity;j++){
	    tab[0][j]=0;
	}
	
	
	for(int i=0;i<=n;i++){
	    tab[i][0]=0;
	}
	
	for(int i=1;i<=n;i++){
	    for(int j=1;j<=capacity;j++){
	        
	        if(weights[i-1]<=j){
	            tab[i][j]=max(value[i-1]+tab[i][j-weights[i-1]],
	                            tab[i-1][j]);
	        }else{
	            tab[i][j]=tab[i-1][j];
	        }
	        
	    }
	}
	
	cout<<tab[n][capacity];
	

}
