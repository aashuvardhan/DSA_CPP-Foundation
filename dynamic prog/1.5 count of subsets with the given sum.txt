	vector<vector<int>> result(n+1,vector<int>(sum+1,-1));
	
	for(int i=0;i<=n;i++){
	    result[i][0]=1;
	}
	
	for(int j=1;j<=sum;j++){
	    result[0][j]=0;
	}
	
	for(int i=1;i<=n;i++){
	    for(int j=1;j<=sum;j++){
	        
	        if(v[i-1]<=j){
	            result[i][j]=result[i-1][j-v[i-1]] + result[i-1][j];
	        }else{
	            result[i][j]=result[i-1][j];
	        }
	        
	    }
	}
	
	cout<<result[n][sum];
	
	

}
