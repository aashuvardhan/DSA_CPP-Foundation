#include <bits/stdc++.h>
using namespace std;


struct Treenode{
    int data;
    Treenode* left;
    Treenode* right;
    
    Treenode(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
    
};


Treenode* conversion(vector<int>& v,int low,int high){
    
    if(low>high){
        return NULL;
    }
    int mid=(low+high)/2;
    
    Treenode* root=new Treenode(v[mid]);
    
    root->left=conversion(v,low,mid-1);
    root->right=conversion(v,mid+1,high);
    return root;
    
    
    
}


void inorder(Treenode* root){
    if(root==NULL){
        return;
    }
    
    inorder(root->left);
    cout<<root->data<<"  ";
    inorder(root->right);
    
}


int main() {
	// your code goes here
	vector<int> v={-10,-3,0,5,9};
	int n=v.size();
	
	Treenode* root=conversion(v,0,n-1);
	
	inorder(root);

}
