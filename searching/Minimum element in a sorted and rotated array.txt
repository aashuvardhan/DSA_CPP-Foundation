'''

Output: 1

Example 2

Input:

Your Task:
Complete the function findMin() which takes an array arr[] and n,

Expected Time Complexity: O(log N).
Expected Auxiliary Space: O(log N).

'''

	int low=0;
	int high=arr.size()-1;
	
	bool flag=false;
	
	 while(low<high){
	     int mid=(low+high)/2;
	     
	     if(mid<high and arr[mid+1]<arr[mid]){
	         cout<<arr[mid+1];
	         flag=true;
	         break;
	     }
	     if(mid>low and arr[mid]<arr[mid-1]){
	         cout<<arr[mid];
	         flag=true;
	         break;
	     }
	     
	     if(arr[mid]<arr[high]){
	         high=mid-1;
	     }else{
	         low=mid+1;
	     }
	     
	 }
	 if(!flag){
	     cout<<arr[low];
	 }
	 

}
