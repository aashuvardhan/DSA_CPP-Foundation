'''

Example 1:

#include <bits/stdc++.h>
using namespace std;

bool checker(vector<int>& arr,int pos,int n){
    
    int count=0;
    for(int i=0;i<arr.size();i++){
        /*
        using the formula of sum of AP and here the sum is the 'pos' variable 
        according to this question the value of a and d is arr[i].
        
        we use the formula s=(n/2)[2a+(n-1)d].
        here we need to find the value of n and here n(from the eq above) gives the number of cakes that has been prepared with th given sum.
        solve the quad eq and ultimately reach the eq:-
        
        n=[-1+sqrt(1+ 8sd )]/2
        
        */
        int val=(-1+sqrt(1+(8*pos/arr[i])))/2;
        cout<<val;
        count+=val;
    }
    return count>=n;
    
}


int main() {
	// your code goes here
	vector<int> v={1, 2, 3, 4};
	
	// here low and high denotes the time range
	int low=0;
	int high=1000000;
	int n=10;
	
	int res=-1;
	
	while(low<=high){
	    int mid=(low+high)/2;
	    
	    if(checker(v,mid,n)){
	        res=mid;
	        high=mid-1;
	        cout<<res<<endl;
	    }else{
	        low=mid+1;
	    }
	    
	    
	}
	
	cout<<res;
	

}
