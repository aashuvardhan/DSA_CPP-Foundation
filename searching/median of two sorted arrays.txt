    vector<int> v1={10,20,30,40,50};
    vector<int> v2={5,15,25,35,45};
    
    int n1=v1.size();
    int n2=v2.size();
    
    int low=0;
    int high=n1-1;
    
    while(low<=high){
        int i1=(low+high)/2;
        int i2=((n1+n2+1)/2)-i1;

		/*

		max1: The largest element on the left side of v1's cut.

		min1: The smallest element on the right side of v1's cut.

		max2: The largest element on the left side of v2's cut.

		min2: The smallest element on the right side of v2's cut.

		*/
        
        int min1=(i1==n1)?INT_MAX:v1[i1];
        int max1=(i1==0)?INT_MIN:v1[i1-1];
        
        int min2=(i2==n2)?INT_MAX:v2[i2];
        int max2=(i2==0)?INT_MIN:v2[i2-1];
        
        if(max1<=min2 && max2<=min1){
            if((n1+n2)%2==0){
                cout<<((double)(max(max1,max2)+min(min1,min2)))/2;
            }else{
                cout<<max(max1,max2);
            }
            break;
        }else if(max1>min2){
            high=i1-1;
        }else{
            low=i1+1;
        }
    
    }
    
}
