#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};



bool symmetric(Node* root1,Node* root2){
    
    if(root1==NULL and root2==NULL){
        return true;
    }else if(root1!=NULL and root2==NULL){
        return false;
    }else if(root1==NULL and root2!=NULL){
        return false;
    }
    
    return (root1->data==root2->data and symmetric(root1->left,root2->right) and symmetric(root1->right,root2->left));
    
    
}



int main() {
	// your code goes here
	


    Node* root=new Node(1);
    root->left=new Node(2);
    root->right=new Node(2);
    root->left->left=new Node(3);
    root->right->right=new Node(3);
    
    root->left->right=new Node(4);
    root->right->left=new Node(4);

    cout<<symmetric(root->left,root->right)<<endl;

}
