https://leetcode.com/problems/longest-zigzag-path-in-a-binary-tree/

*/
#include <bits/stdc++.h>
using namespace std;

struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


struct Triplet{
    Node* root;
    bool isLeft;
    int depth;
    Triplet(Node* rt, bool Left, int dp){
        root=rt;
        isLeft=Left;
        depth=dp;
    }
};



int main() {
	// your code goes here

    Node* root=new Node(1);
    root->left=new Node(1);
    root->right=new Node(1);
    
    root->left->right=new Node(1);
    root->left->right->left=new Node(1);
    root->left->right->left->right=new Node(1);
    
    root->left->right->right=new Node(1);
    
    
    vector<Triplet* > queue;
    
    int res=0;
    
    if(root->left){
        Triplet* t=new Triplet(root->left,true,1);
        queue.push_back(t);
    }
    if(root->right){
        Triplet* t=new Triplet(root->right,false,1);
        queue.push_back(t);
    }
    
    while(queue.size()!=0){
        Triplet* t=*queue.begin();
        queue.erase(queue.begin());
        
        Node* curr=t->root;
        bool isLeft=t->isLeft;
        int depth=t->depth;
        
        res=max(res,depth);
        
        if(isLeft==true){
            if(curr->right!=NULL){
                queue.push_back(new Triplet(curr->right,false,depth+1));
            }
            if(curr->left!=NULL){
                queue.push_back(new Triplet(curr->left,true,1));
            }
        }else{
            if(curr->right!=NULL){
                queue.push_back(new Triplet(curr->right,false,1));
            }
            if(curr->left!=NULL){
                queue.push_back(new Triplet(curr->left,true,depth+1));
            }
        }
    }
    
    cout<<res;


}
