#include <bits/stdc++.h>
using namespace std;

struct Node{
    int data;
    Node* left;
    Node* right;
    
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


int main() {
	// your code goes here
	
    Node* root=new Node(10);
    root->left=new Node(8);
    root->right=new Node(2);
    
    root->left->left=new Node(3);
    root->left->right=new Node(5);
    
    root->right->left=new Node(2);
    
    
    // start like level order traversal
    vector<Node*> queue;
    queue.push_back(root);
    while(queue.size()!=0){
        int count=queue.size();
        int total=0;
        for(int i=0;i<count;i++){
            Node* curr=*queue.begin();
            queue.erase(queue.begin());
            
            while(curr!=NULL){
                total+=curr->data;
                if(curr->left!=NULL){
                    queue.push_back(curr->left);
                }
                curr=curr->right;
            }
        }
        cout<<total<<endl;
    }
    
}
