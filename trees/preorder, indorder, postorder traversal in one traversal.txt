#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};



int main() {
	// your code goes here


    Node* root=new Node(1);
    root->left=new Node(2);
    root->left->left=new Node(3);
    root->left->right=new Node(4);
    
    root->right=new Node(5);
    root->right->left=new Node(6);
    root->right->right=new Node(7);
    
    
    vector<int> preorder;
    vector<int> inorder;
    vector<int> postorder;
    vector<pair<Node*, int>>stack;
    
    stack.push_back({root,1});
    while(stack.size()!=0){
        pair<Node*,int> p=*(stack.end()-1);
        stack.pop_back();
        Node* curr=p.first;
        int val=p.second;
        
        if(val==1){
            preorder.push_back(curr->data);
            val++;
            stack.push_back({curr,val});
            if(curr->left!=NULL){
                stack.push_back({curr->left,1});
            }
        }else if(val==2){
            inorder.push_back(curr->data);
            val++;
            stack.push_back({curr,val});
            if(curr->right!=NULL){
                stack.push_back({curr->right,1});
            }
        }else{
            postorder.push_back(curr->data);
        }
        
        
    }
    
    
    cout<<"Preorder:- ";
    for(auto ele: preorder){
        cout<<ele<<" ";
    }
    cout<<endl;
    
    cout<<"Inorder:- ";
    for(auto ele: inorder){
        cout<<ele<<" ";
    }
    cout<<endl;
    
    cout<<"Postorder:- ";
    for(auto ele: postorder){
        cout<<ele<<" ";
    }
    cout<<endl;
}
