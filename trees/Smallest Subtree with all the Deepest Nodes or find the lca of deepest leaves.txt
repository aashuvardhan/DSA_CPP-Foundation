#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        left=NULL;
        right=NULL;
        data=x;
    }
};


vector<int> parser(string str){
    vector<int> ans;
    string strg="";
    for(int i=0;i<str.size();i++){
        if(str[i]!='#'){
            strg+=str[i];
        }else{
            int val=stoi(strg);
            ans.push_back(val);
            strg="";
        }
    }
    int val=stoi(strg);
    ans.push_back(val);
    return ans;
}



void traversal(Node* root,vector<Node*> v, unordered_map<int,vector<vector<Node*>>>& umap,int depth){
    
    if(root==NULL){
        return;
    }
    
    
    
    if(root->left==NULL and root->right==NULL){
        v.push_back(root);
        umap[depth].push_back(v);
        v.pop_back();
        return;
    }
    
    v.push_back(root);
    
    traversal(root->left,v,umap,depth+1);
    traversal(root->right,v,umap,depth+1);
    
    v.pop_back();
    
    
}

void inorder(Node* root){
    if(root==NULL){
        return;
    }
    inorder(root->left);
    cout<<root->data<<endl;
    inorder(root->right);
}


int main() {
	// your code goes here
	

    Node* root=new Node(3);
    root->left=new Node(5);
    root->left->left=new Node(6);
    root->left->right=new Node(2);
    
    root->left->right->left=new Node(7);
    root->left->right->right=new Node(4);
    
    root->right=new Node(1);
    root->right->right=new Node(8);
    root->right->left=new Node(0);
    
    // step1 :- find the deepest nodes
    
    unordered_map<int,vector<vector<Node*>>> umap;
    
    traversal(root,{},umap,0);
    
    int maxi=0;
    
    for(auto ele: umap){
        maxi=max(maxi,ele.first);
    }
    
    
    
    //step 2 :- find the lca
    Node* res=NULL;
    for(int i=0;i<=maxi;i++){
        Node* first=umap[maxi][0][i];
        bool flag=false;
        for(int j=1;j<umap[maxi].size();j++){
            
            if(umap[maxi][j][i]->data!=first->data){
                flag=true;
                break;
            }
            
            
        }
        if(flag){
            break;
        }else{
            res=first;
        }
    }
    
    if(res!=NULL){
        inorder(res);
    }
    
    
}
