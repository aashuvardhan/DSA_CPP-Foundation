#include <bits/stdc++.h>
using namespace std;

struct Node {
    int data;
    Node* left;
    Node* right;
    Node(int x) : data(x), left(NULL), right(NULL) {}
};

// Returns a pair: {maximum depth, LCA node of deepest leaves in this subtree}
pair<int, Node*> getLCA(Node* root, int depth) {
    if (root == NULL) {
        return {depth, NULL};
    }
    
    pair<int, Node*> leftResult = getLCA(root->left, depth + 1);
    pair<int, Node*> rightResult = getLCA(root->right, depth + 1);
    
    // If left and right are at the same max depth, this root is the LCA
    if (leftResult.first == rightResult.first) {
        return {leftResult.first, root};
    }
    
    // Otherwise, pass up the side that goes deeper
    if (leftResult.first > rightResult.first) {
        return leftResult;
    } else {
        return rightResult;
    }
}

void inorder(Node* root){
    if(root==NULL){
        return;
    }
    inorder(root->left);
    cout<<root->data<<"  ";
    inorder(root->right);
}

int main() {
    // Tree construction
    Node* root = new Node(3);
    root->left = new Node(5);
    root->left->left = new Node(6);
    root->left->right = new Node(2);
    root->left->right->left = new Node(7);
    root->left->right->right = new Node(4);
    root->right = new Node(1);
    root->right->left = new Node(0);
    root->right->right = new Node(8);
    
    pair<int, Node*> ans = getLCA(root, 0);
    
    if (ans.second != NULL) {
        cout << "LCA Data: " << ans.second->data << endl; 
    }
    inorder(ans.second);
    
    return 0;
}
