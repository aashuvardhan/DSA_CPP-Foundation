/*

https://leetcode.com/problems/all-nodes-distance-k-in-binary-tree/


Given the root of a binary tree, the value of a target node target, and an integer k, return an array of the values of all nodes that have a distance k from the target node.

You can return the answer in any order.

*/



#include <bits/stdc++.h>
using namespace std;

struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


void dfs(unordered_map<int,vector<int>>& graph,int source,int k,unordered_set<int>& st,vector<int>& ans){
    st.insert(source);
    
    if(source==-1){
        return;
    }
    if(k==0){
        ans.push_back(source);
        return;
    }
    
    for(auto ele: graph[source]){
        if(st.find(ele)==st.end()){
            dfs(graph,ele,k-1,st,ans);
        }
    }
}



void traversal(Node* root,unordered_map<int,vector<int>>& graph,int parent){
    if(root==NULL){
        return;
    }
    
    traversal(root->left,graph,root->data);
    
    graph[parent].push_back(root->data);
    graph[root->data].push_back(parent);
    
    traversal(root->right,graph,root->data);
}




int main() {
	// your code goes here


    Node* root= new Node(3);
    root->left=new Node(5);
    root->left->left=new Node(6);
    root->left->right=new Node(2);
    root->left->right->left=new Node(7);
    
    root->left->right->right=new Node(4);
    root->right=new Node(1);
    root->right->right=new Node(8);
    root->right->left=new Node(0);
    
    int target=5;
    int k=2;
    
    
    // creating graph out of the tree
    
    unordered_map<int,vector<int>> graph;
    traversal(root,graph,-1);
    
    // traverse the graph with the source as target and keep the record of all the nodes at k distance.
    vector<int> ans;
    unordered_set<int> st;
    
    dfs(graph,target,k,st,ans);
    
    for(auto ele: ans){
        cout<<ele<<"  ";
    }
    
    
    

}
