#include <bits/stdc++.h>
using namespace std;

struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};

struct ListNode{
    int data;
    ListNode* prev;
    ListNode* next;
    ListNode(int x){
        data=x;
        prev=NULL;
        next=NULL;
    }
};

ListNode* previous=NULL;
ListNode* head=NULL;

void traversal(Node* root){
    if(root==NULL){
        return;
    }
    traversal(root->left);
    
    ListNode* temp=new ListNode(root->data);
    if(previous==NULL){
        head=temp;
    }else{
        temp->prev=previous;
        previous->next=temp;
    }
    previous=temp;
    traversal(root->right);
    
}

void list_traversal(ListNode* head){
    ListNode* curr=head;
    
    while(curr!=NULL){
        cout<<curr->data<<"  ";
        curr=curr->next;
    }
}



int main() {

    Node* root=new Node(10);
    root->left=new Node(5);
    root->right=new Node(20);
    root->right->left=new Node(30);
    root->right->right=new Node(35);
    
    traversal(root);
    
    list_traversal(head);
	

}
