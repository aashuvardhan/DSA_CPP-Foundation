#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


Node* traversal(Node* root1,Node* root2){
    
    if(root1==NULL and root2!=NULL){
        root1=new Node(root2->data);
        root1->left=traversal(NULL,root2->left);
        root1->right=traversal(NULL,root2->right);
    }else if(root1!=NULL and root2==NULL){
        root1->left=traversal(root1->left,NULL);
        root1->right=traversal(root1->right,NULL);
    }else if(root1!=NULL and root2!=NULL){
        root1->data+=root2->data;
        root1->left=traversal(root1->left,root2->left);
        root1->right=traversal(root1->right,root2->right);
    }
    return root1;
}


void inorder(Node* root){
    if(root==NULL){
        return;
    }
    inorder(root->left);
    cout<<root->data<<endl;
    inorder(root->right);
}


int main() {
	// your code goes here
	
	

    Node* root1=new Node(1);
    root1->left=new Node(3);
    root1->left->left=new Node(5);
    root1->right=new Node(2);
    
    

    Node* root2=new Node(2);
    root2->left=new Node(1);
    root2->left->right=new Node(4);
    root2->right=new Node(3);
    root2->right->right=new Node(7);
    
    Node* root;
    
    if(root1==NULL and root2==NULL){
        root=NULL;
    }else{
        root=traversal(root1,root2);
        inorder(root);
    }
    

}
