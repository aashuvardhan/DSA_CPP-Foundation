#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


int max_Sum_leaf_nodes(Node* root,int *ptr){
    if(root==NULL){
        return 0;
    }
    if(root->left==NULL and root->right==NULL){
        return root->data;
    }
    
    int lv=max_Sum_leaf_nodes(root->left,ptr);
    int rv=max_Sum_leaf_nodes(root->right,ptr);
    
    *ptr=max(*ptr,root->data+lv+rv);
    return max(lv,rv)+root->data;
}



int main() {
	// your code goes here

    Node* root=new Node(10);
    root->left=new Node(-9);
    root->right=new Node(20);
    root->right->left=new Node(-15);
    root->right->right=new Node(-7);
    
    
    int res=INT_MIN;
    max_Sum_leaf_nodes(root,&res);
    cout<<res;

}
