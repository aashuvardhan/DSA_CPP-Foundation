#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


int count_complete_node(Node* root){
    int lh=0;
    int rh=0;
    Node* curr=root;
    
    while(curr!=NULL){
        lh++;
        curr=curr->left;
    }
    
    curr=root;
    while(curr!=NULL){
        rh++;
        curr=curr->right;
    }
    
    if(lh==rh){
        return pow(2,lh)-1;
    }else{
        return 1+count_complete_node(root->left)+count_complete_node(root->right);
    }
}

    root->left->left=new Node(40);
    root->left->right=new Node(50);
    root->left->left->left=new Node(80);
    root->left->left->right=new Node(90);
    
    root->left->right=new Node(50);
    root->left->right->left=new Node(100);
    root->left->right->right=new Node(110);
    
    
    cout<<count_complete_node(root);

}
