#include <bits/stdc++.h>
using namespace std;

struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};


void traversal(Node* root,int *ptr,string str){
    if(root==NULL){
        return;
    }
    
    str+=to_string(root->data);
    if(root->left==NULL and root->right==NULL){
        *ptr+=stoi(str,0,2);
        return;
    }
    traversal(root->left,ptr,str);
    traversal(root->right,ptr,str);
}



int main() {
	// your code goes here
	
	
    
    Node* root=new Node(1);
    root->right=new Node(1);
    root->right->right=new Node(1);
    root->right->left=new Node(0);
    
    root->left=new Node(0);
    root->left->left=new Node(0);
    root->left->right=new Node(1);
    
    int res=0;
    traversal(root,&res,"");
    cout<<res;
	

}
