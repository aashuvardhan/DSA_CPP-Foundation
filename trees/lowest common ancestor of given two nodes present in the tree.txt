#include <bits/stdc++.h>
using namespace std;

struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};

Node* lowestCommonAncestor(Node* root, int n1,int n2){
    
    if(root==NULL or root->data==n1 or root->data==n2){
        return root;
    }
    
    Node* left=lowestCommonAncestor(root->left,n1,n2);
    Node* right=lowestCommonAncestor(root->right,n1,n2);
    
    if(left!=NULL and right!=NULL){
        return root;
    }
    else if(left!=NULL){
        return left;
    }else{
        return right;
    }
    
    
}



int main() {
	// your code goes here
	Node* root=new Node(20);
	root->left=new Node(12);
	root->right=new Node(8);
	root->right->left=new Node(3);
	root->right->right=new Node(5);
	
	cout<<lowestCommonAncestor(root,12,5)->data<<endl;
	
}
