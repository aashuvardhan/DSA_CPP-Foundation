#include <bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    Node* left;
    Node* right;
    Node(int x){
        data=x;
        left=NULL;
        right=NULL;
    }
};



void traversal(Node* root){
    if(root==NULL){
        return;
    }
    traversal(root->left);
    cout<<root->data<<"  ";
    traversal(root->right);
}

Node* invert(Node* root){
    if(root==NULL){
        return NULL;
    }
    Node* L=invert(root->left);
    Node* R=invert(root->right);
    
    Node* temp=L;
    root->left=R;
    root->right=temp;
    return root;
}




int main() {
	// your code goes here
	
	

    Node* root=new Node(4);
    root->left=new Node(2);
    root->left->left=new Node(1);
    root->left->right=new Node(3);
    root->right=new Node(7);
    root->right->right=new Node(9);
    root->right->left=new Node(6);
    
    traversal(root);
    cout<<endl;
    invert(root);
    traversal(root);

}
