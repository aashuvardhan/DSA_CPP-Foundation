/*
Given an array arr[] of N Numbers. A Perfect Piece is defined as a subarray such that the difference between the minimum and the maximum value in that range is at most 1. Find the Maximal Length Perfect Piece.

 

Example 1:

Input:
N = 4
arr[] = {8, 8, 8, 8}
Output:
4
Explanation:
The longest subarray is [1, 4]
where maximum=8 and minimum = 8 and
difference is 0, which is less than 1.
Its length is 4.
Example 2:

Input:
N = 11
arr[] = {5, 4, 5, 5, 6, 7, 8, 8, 8, 7, 6}
Output:
5
Explanation:
The longest subarray is [6, 10]
where maximum=8 and minimum = 7 and
difference is 1. Its length is 5. 
 

Your Task:
You don't need to read input or print anything. Your task is to complete the function longestPerfectPiece() which takes an Integer N and an array arr[] of length N as input and returns the maximal length Perfect Piece.

 

Expected Time Complexity: O(N*logN)
Expected Auxiliary Space: O(N)




*/
#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	vector<int> v={5, 4, 5, 5, 6, 7, 8, 8, 8, 7, 6};
	
	int n=v.size();
	int i=0;
	int j=1;
	int res=1;
	
	vector<int> currmin={0};
	vector<int> currmax={0};
	
	while(j<n){
	    
	    while(currmin.size()!=0 and currmin[0]<i){
	        currmin.erase(currmin.begin());
	    }
	    
	    while(currmax.size()!=0 and currmax[0]<i){
	        currmax.erase(currmax.begin());
	    }
	    
	    while(currmin.size()!=0){
	        int lastele=v[*(currmin.rbegin())];
	        if(lastele>v[j]){
	            currmin.pop_back();
	        }else{
	            break;
	        }
	    }
	    currmin.push_back(j);
	    
	    while(currmax.size()!=0){
	        int lastele=v[*(currmax.rbegin())];
	        if(lastele<v[j]){
	            currmax.pop_back();
	        }else{
	            break;
	        }
	    }
	    
	    currmax.push_back(j);
	    
	    if((v[currmax[0]]-v[currmin[0]])>1){
	        i=max(currmax[0],currmin[0])+1;
	    }else{
	        res=max(res,j-i+1);
	    }
	    j+=1;
	    
	    
	}
	cout<<res;
	

}
