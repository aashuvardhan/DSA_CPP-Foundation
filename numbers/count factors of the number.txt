#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	
	int n=59;
	
	int i=2;
	int res=1;
	while(i*i<=n){
	    if(n%i==0){
	        int count=0;
	        while(n%i==0){
	            n=n/i;
	            count++;
	        }
	        res=res*(count+1);
	    }
	    i++;
	}
	if(n>1){
	    res=res*(1+1);
	}
	cout<<(res);

}
