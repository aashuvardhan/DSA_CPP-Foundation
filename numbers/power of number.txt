#include <bits/stdc++.h>
using namespace std;

int power(int n,int p){
    if(p==0){
        return 1;
    }
    int temp=power(n,p/2);
    temp=temp*temp;
    if(p%2==0){
        return temp;
    }else{
        return temp*n;
    }
}
