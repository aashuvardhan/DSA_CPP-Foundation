#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;
	cin>>n;
	
	vector<int> arr;
	
	vector<int> rem_arr(n,-1);
	
	for(int i=0;i<n;i++){
	    int m;
	    cin>>m;
        arr.push_back(m);	    
	}
	int sum=arr[0];
	
	int rem=sum%n;
	//cout<<"rem: "<<rem<<endl;
	if(rem==0){
	    cout<<1<<"=="<<1<<endl;
	}else{
	    rem_arr[rem]=1;
	    
	}
    for(int i=1;i<n;i++){
        sum+=arr[i];
        
        rem=sum%n;
        //cout<<"rem: "<<rem<<endl;
        if(rem==0){
            cout<<1<<"=="<<i+1<<endl;
        }else if(rem_arr[rem]==-1){
            rem_arr[rem]=i;
        }else{
            cout<<rem_arr[rem]+1<<"=="<<i+1<<endl;
        }
    }
	

}
