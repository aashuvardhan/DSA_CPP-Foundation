#include <bits/stdc++.h>
using namespace std;


string palindromic(string str,int left,int right){
    
    while(left>=0 and right<str.size()){
        if(str[left]!=str[right]){
            break;
        }
        left--;
        right++;
    }
    
    return str.substr(left+1,right-left-1);
}


int main() {
	// your code goes here
	string str="babad";
	
	string res="";
	
	for(int i=0;i<str.size();i++){
	    string val1=palindromic(str,i,i);//odd
	    string val2=palindromic(str,i,i+1);//even
	    
	    if(val1.size()>=val2.size()){
	        if(val1.size()>res.size()){
	            res=val1;
	        }
	    }else{
	        if(val2.size()>res.size()){
	            res=val2;
	        }
	    }
	}
	cout<<res;
	
}
